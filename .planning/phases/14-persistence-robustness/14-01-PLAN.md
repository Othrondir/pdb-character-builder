Close ROADMAP SC#1 (Phase 14): the global toast in `apps/planner/src/components/ui/toast.tsx` currently overwrites `current` on every `pushToast` call. A second `pushToast` fired within 100 ms (e.g., chained save → success → version-warning) clobbers the first message before the user can read it.

    ```typescript
    export function pushToast(body: string, tone: ToastMessage['tone'] = 'info'): void {
      const next: ToastMessage = { id: nextId++, body, tone };
      const now = Date.now();
      const elapsed = current === null ? Infinity : now - visibleSince;
      if (current === null || elapsed >= MIN_VISIBLE_MS) {
        current = next;
        visibleSince = now;
        listeners.forEach((fn) => fn(current));
        return;
      }
      queue.push(next);
      scheduleDrain();
    }
    ```

    Step 3: Replace the body of `dismissToast` (lines 23-26) with:
    ```typescript
    export function dismissToast(): void {
      if (drainTimer !== null) {
        clearTimeout(drainTimer);
        drainTimer = null;
      }
      const head = queue.shift();
      if (head !== undefined) {
        current = head;
        visibleSince = Date.now();
        listeners.forEach((fn) => fn(current));
        scheduleDrain();
        return;
      }
      current = null;
      listeners.forEach((fn) => fn(null));
    }
    ```

    ```typescript
    function scheduleDrain(): void {
      if (queue.length === 0 || drainTimer !== null) return;
    ```
