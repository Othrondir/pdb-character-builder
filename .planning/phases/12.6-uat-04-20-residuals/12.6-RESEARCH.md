```typescript
// packages/rules-engine/src/foundation/point-buy-snapshot.ts (NEW)
import { z } from 'zod';
import type { CanonicalId } from '../contracts/canonical-id';
import snapshotJson from './data/puerta-point-buy.json';

const canonicalRaceIdRegex = /^race:[A-Za-z0-9._-]+$/;

export const pointBuyCurveSchema = z.object({
  budget: z.number().int().positive(),
  minimum: z.number().int(),
  maximum: z.number().int(),
  costByScore: z.record(
    z.string().regex(/^\d+$/),
    z.number().int().nonnegative(),
  ),
}).refine((c) => c.minimum <= c.maximum, {
  message: 'minimum must be <= maximum',
}).refine(
  (c) => Object.keys(c.costByScore).every(
    (k) => Number(k) >= c.minimum && Number(k) <= c.maximum,
  ),
  { message: 'costByScore keys must be within [minimum, maximum]' },
);

export const pointBuySnapshotSchema = z.record(
  z.string().regex(canonicalRaceIdRegex),
  pointBuyCurveSchema,
);

export type PointBuyCurve = z.infer<typeof pointBuyCurveSchema>;
export type PointBuySnapshot = z.infer<typeof pointBuySnapshotSchema>;

// Fail-closed at module load. Any schema mismatch throws before the UI mounts.
export const PUERTA_POINT_BUY_SNAPSHOT: Record<CanonicalId, PointBuyCurve> =
  pointBuySnapshotSchema.parse(snapshotJson) as Record<CanonicalId, PointBuyCurve>;
```

### Pattern 5: Level legality derivation (G1 + 12.4-09 reuse)

```typescript
// apps/planner/src/features/level-progression/selectors.ts (EDIT)

export type LevelLegality = 'legal' | 'incomplete' | 'invalid' | 'locked';

/**
 * Phase 12.6 P5 — D-17 legality classification.
 *
 * Reuses existing G1 sequential-gate predicate (prior-level-has-class)
 * and 12.4-09's selectLevelCompletionState. Precedence (locked wins):
 *   locked   — prior level incomplete (G1)
 *   invalid  — level has illegal issues in the revalidated progression
 *   incomplete — missing feat or skill choices (LevelCompletionState)
 *   legal    — otherwise
 */
export function selectLevelLegality(
  progressionState: LevelProgressionStoreState,
  foundationState: CharacterFoundationStoreState,
  featState: FeatStoreState,
  skillState: SkillStoreState,
  level: ProgressionLevel,
): LevelLegality {
  // G1 locked gate: rebuild the rail-style predicate in one pass.
  const rail = selectLevelRail(progressionState, foundationState);
  const entry = rail.find((r) => r.level === level);
  if (entry?.locked) return 'locked';
  if (entry?.status === 'illegal') return 'invalid';

  const completion = selectLevelCompletionState(
    progressionState, foundationState, featState, skillState, level,
  );
  if (!completion.isComplete) return 'incomplete';

  return 'legal';
}
```

**Reuse gates:**

- `selectLevelRail` already emits `locked` via the G1 predicate `prevHasClass` tracker (see `selectors.ts:180-188`).
- `selectLevelCompletionState` already exposes `featDeficit` + `skillDeficit` + `isComplete` (see `selectors.ts:513-538`).
- No new pure helper is strictly necessary, but a named `selectLevelLegality` exported selector gives tests a stable anchor + prevents row components from re-deriving.

### Anti-Patterns to Avoid

- **Row-local `useState` for expansion.** Derive from `progressionState.activeLevel === level`; local state causes duplicate state + fights atomic dispatch.
- **Inline Zod parsing inside render.** Must happen at module load so failures surface at startup, not per-render.
- **Adding a `verified` boolean to the JSON schema** (CONTEXT rejected). Provenance lives in the sibling `.md`.
- **Mounting `<ClassPicker />` inside every row.** SPEC D-11 + performance — only the active row mounts heavy children. 20 mounts × class-picker reachable-at-level-N computation per row = pathological re-renders.
- **`aria-disabled` without `disabled` on locked rows.** Screen readers follow the native `disabled` attr to prevent activation; UI-SPEC D-10 requires both.
- **Deleting `level-rail.tsx` before scrubbing `creation-stepper.tsx`.** Type error breaks the whole build.
- **Bumping `buildDocumentSchema.schemaVersion` to accommodate a `pointBuyCurveVersion` field.** SHAR-05 invariant — verified by `grep "schemaVersion" packages/rules-engine/src/share-url/` stays at baseline.
- **Dropping the `DetailPanel` / `NwnFrame` primitives inside the expanded row.** Host-swap means the internals render unchanged; do not re-shell the contents.

## Common Pitfalls

### Pitfall 1: Dangling `<LevelRail />` in `creation-stepper.tsx` after `level-rail.tsx` delete

**What goes wrong:** `tsc` breaks the build with "Cannot find module './level-rail'". Dev server crashes. Vitest runs fail at import resolution before any test executes.

**Why it happens:** CONTEXT D-16 says delete the file; only the `build-progression-board.tsx` and `level-sheet.tsx` scan-surface consumers are called out explicitly. The `creation-stepper.tsx` shell sidebar mounts `<LevelRail />` independently to drive per-level navigation outside the main content area.

**How to avoid:** Plan the deletion as a single atomic task that (a) removes the JSX from `creation-stepper.tsx:134`, (b) removes the import at `creation-stepper.tsx:14`, (c) decides what replaces it (options: render nothing, render a tiny "L{activeLevel}" chip, render the new `<LevelProgressionRow />` list — but that creates double mounting), and (d) deletes `level-rail.tsx` + its CSS rules. Commit the four changes together.

**Warning signs:** `grep -rn "level-rail\|LevelRail" apps/planner/src/` after the plan returns anything outside `styles/app.css`.

### Pitfall 2: Expanded row mounts on non-`class` sub-step

**What goes wrong:** User clicks Habilidades chip → `activeLevelSubStep = 'skills'` → `center-content.tsx` routes to `<SkillBoard />`. But `<BuildProgressionBoard />` is no longer rendered, so the expanded row is irrelevant. Conversely, if the expanded row renders `<ClassPicker />` + `<LevelEditorActionBar />` even when `activeLevelSubStep !== 'class'`, state dispatch inside those components becomes inconsistent with the visible surface.

**Why it happens:** SPEC R6 + D-11 describe "click row → expand inline". The hidden assumption is `activeLevelSubStep === 'class'`; the active cursor is gatekept externally by `center-content.tsx`.

**How to avoid:** The expanded slot render predicate is already `progressionState.activeLevel === N` — this is safe because `<BuildProgressionBoard />` is only visible on the `class` sub-step route. Do NOT add a `&& activeLevelSubStep === 'class'` guard; center-content gating is sufficient. Document this in the component comment so future refactorers don't mis-derive.

**Warning signs:** If an RTL spec directly renders `<LevelProgressionRow />` without setting `usePlannerShellStore` sub-step, the expanded slot still renders but dispatch wiring is meaningful. Test fixtures must set `activeLevelSubStep = 'class'` explicitly.

### Pitfall 3: `scrollHeight === clientHeight` RTL assertion hangs on jsdom

**What goes wrong:** Under `@vitest-environment jsdom`, `container.scrollHeight` and `container.clientHeight` both return 0. The equality is trivially true → the assertion passes **regardless** of whether the 20 rows would actually fit in a real 800px viewport. Worse: RED/GREEN cycles never flip red on layout regressions.

**Why it happens:** jsdom has no layout engine (documented jsdom issue #1013). Phase 12.4-02 already encountered this and routed to a hybrid CSS-text + DOM-class + `getComputedStyle` approach.

**How to avoid:** Use the phase-12.4 pattern (see Pattern 4 above). Assert CSS rule presence via text grep on `app.css`, assert the 20 `<li>` elements via DOM query, and assert `getComputedStyle(list).gridTemplateRows.includes('repeat(20')` after injecting `app.css`. SPEC R5's literal text should be softened in the plan's acceptance criteria.

**Warning signs:** A test that asserts `scrollHeight === clientHeight` passes on master without the new code — instant false-positive.

### Pitfall 4: `canonicalRaceIdSchema` does not exist as an exported symbol

**What goes wrong:** CONTEXT D-02 says `z.record(canonicalRaceIdSchema, pointBuyCurveSchema)`. The implementer searches for `canonicalRaceIdSchema`, finds zero exports (Grep-verified), and either (a) wastes time asking or (b) defines an ad-hoc one.

**Why it happens:** CONTEXT.md was authored referencing a logical schema name that doesn't exist in code. The closest existing symbol is the inline regex `z.string().regex(/^race:[A-Za-z0-9._-]+$/)` at `packages/data-extractor/src/contracts/race-catalog.ts:13`.

**How to avoid:** Plan either (a) to define `canonicalRaceIdSchema` in `packages/rules-engine/src/contracts/canonical-id.ts` as `export const canonicalRaceIdSchema = z.string().regex(/^race:[A-Za-z0-9._-]+$/)` and export it (cleaner, reusable) OR (b) inline the regex literal inside `point-buy-snapshot.ts` (minimum surface). Option (a) adds three lines but resolves a known gap.

**Warning signs:** TypeScript error `Cannot find name 'canonicalRaceIdSchema'` in the new `point-buy-snapshot.ts`.

### Pitfall 5: Snapshot race-coverage test flakes due to `race:drow` duplicate

**What goes wrong:** `compiled-races.ts` emits 46 race rows; 45 are unique after `dedupeByCanonicalId`. `race:drow` appears at sourceRow 196 AND 676. A coverage test that does `Object.keys(snapshot).length === compiledRaceCatalog.races.length` would demand 46 entries; one `race:drow` entry suffices for the 45 unique IDs.

**Why it happens:** Extractor emits duplicates; Phase 12.1-02 installed `dedupeByCanonicalId` first-wins at projection boundary. The 12.6-SPEC acceptance correctly says "every race in `dedupeByCanonicalId(compiledRaceCatalog.races)`" — but the implementer must actually call the dedupe helper.

**How to avoid:** In `tests/phase-12.6/point-buy-snapshot-coverage.spec.ts`, compute the expected count via `new Set(compiledRaceCatalog.races.map((r) => r.id)).size` (or import + call `dedupeByCanonicalId`). Assert `Object.keys(snapshot).length === uniqueCount`. Document in the spec comment that extractor-emitted duplicates are dedup-first-wins.

**Warning signs:** Coverage test asserts `=== 46` and fails with "missing race:drow second entry" error.

### Pitfall 6: `rule:point-buy-missing` does not satisfy `canonicalIdRegex`

**What goes wrong:** `resolveValidationOutcome` eventually parses `affectedIds` via `z.string().regex(canonicalIdRegex)` where `canonicalIdRegex = /^[a-z-]+:[A-Za-z0-9._-]+$/`. The ID `rule:point-buy-missing` matches this regex (`rule` is a valid `EntityKind`, `point-buy-missing` satisfies `[A-Za-z0-9._-]+`). Double-check: uppercase letters? none. Colons inside suffix? none. Dots? none. Dashes? yes, legal. **Safe.**

**Why it matters:** Existing rule IDs follow this shape (`rule:origin-incomplete`, `rule:ability-budget`, `rule:ability-str`) — the new ID matches the pattern. `[VERIFIED: canonicalIdRegex + existing rule IDs]`.

**How to avoid:** Verify the regex match once in the RED spec: `expect(canonicalIdRegex.test('rule:point-buy-missing')).toBe(true)`.

### Pitfall 7: `pointBuyMissing` callout is emitted BUT the `disabled` prop isn't added to `-` buttons

**What goes wrong:** Current `AttributesBoard` at `attributes-board.tsx:85-91`: the `-` button's `disabled` prop is `baseValue <= minimum`, independent of the curve. In fail-closed state, `baseValue === minimum === 8`, so `-` is disabled naturally. But if the user's base has been edited higher before the race swap that nulled the curve, `-` would still be clickable. SPEC R3 says all `+` AND `-` buttons disabled.

**Why it happens:** Existing button gate predicates don't know about the snapshot's null branch.

**How to avoid:** In AttributesBoard, thread a local `const isBlocked = attributeBudget.status === 'blocked' && attributeBudget.issues.some((i) => i.affectedIds.includes('rule:point-buy-missing'))`. Disabled `+`: `isBlocked || !canIncrementAttribute(...)`. Disabled `-`: `isBlocked || baseValue <= minimum`. Keeps the two independent states composable.

**Warning signs:** RTL assertion `[aria-label^="Reducir"]:disabled` count < 6 while the callout is visible.

### Pitfall 8: CSS overflow + `overflow-y: auto` on list competes with the expanded-row internal scroll

**What goes wrong:** The UI-SPEC declares `.level-progression__list { overflow-y: auto }` AND `.class-picker__list { max-height: 34vh; overflow-y: auto }` (12.4-06). When an active row expands and its ClassPicker list scrolls, scroll events can bubble and drag the outer list too.

**Why it happens:** Default scroll propagation. The inner overflow catches the scroll at viewport top but releases to the parent at both ends.

**How to avoid:** Either (a) set `overscroll-behavior: contain` on `.level-progression__list` to isolate scroll, or (b) only render `overflow-y: auto` on the outer list when the expanded row overflows (CSS media query approach — brittle). Option (a) is zero-cost: one property, standards-supported since 2018.

**Warning signs:** UAT verbal feedback about "weird scroll jumps" when scrolling inside the class picker.
