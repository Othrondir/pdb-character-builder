    export interface LevelCompletionState {
      level: ProgressionLevel;
      featDeficit: number;
      skillDeficit: number;
      isComplete: boolean;
    }

    export function selectLevelCompletionState(
      progressionState: LevelProgressionStoreState,
      foundationState: CharacterFoundationStoreState,
      featState: FeatStoreState,
      skillState: SkillStoreState,
      classCatalog: ClassCatalogInput,
      featCatalog: FeatCatalogInput,
      raceCatalog: RaceCatalogInput,
      level: ProgressionLevel,
    ): LevelCompletionState {
      const snapshot = toBuildSnapshot(progressionState, foundationState, featState, skillState);
      const budget = computePerLevelBudget(snapshot, level, classCatalog, featCatalog, raceCatalog);
      const featDeficit = Math.max(0, budget.featSlots.total - budget.featSlots.chosen);
      const skillDeficit = Math.max(0, budget.skillPoints.budget - budget.skillPoints.spent);
      return {
        level,
        featDeficit,
        skillDeficit,
        isComplete: featDeficit === 0 && skillDeficit === 0 && budget.featSlots.total > 0,
      };
    }

    export function computeAdvanceLabel(state: LevelCompletionState): { label: string; disabled: boolean } {
      const copy = shellCopyEs.progression.advanceButton;
      // Priority: feat deficit > skill deficit (UI-SPEC.md R2 "both unfilled" rule)
      if (state.featDeficit > 0) {
        const template = state.featDeficit === 1 ? copy.deficitFeatsSingular : copy.deficitFeatsPluralTemplate.replace('{N}', String(state.featDeficit));
        return { label: template, disabled: true };
      }
      if (state.skillDeficit > 0) {
        const template = state.skillDeficit === 1 ? copy.deficitSkillsSingular : copy.deficitSkillsPluralTemplate.replace('{N}', String(state.skillDeficit));
        return { label: template, disabled: true };
      }
      return {
        label: copy.continueTemplate.replace('{N}', String(state.level + 1)),
        disabled: false,
      };
    }

    import { useLevelProgressionStore } from './store';
    import { useCharacterFoundationStore } from '../character-foundation/store';
    import { useFeatStore } from '../feats/store';
    import { useSkillStore } from '../skills/store';
    import { NwnButton } from '@planner/components/ui/nwn-button';
    import { compiledClassCatalog } from '@planner/data/compiled-classes';
    import { compiledFeatCatalog } from '@planner/data/compiled-feats';
    import { compiledRaceCatalog } from '@planner/data/compiled-races';
    import { selectLevelCompletionState, computeAdvanceLabel } from './selectors';
    import type { ProgressionLevel } from './progression-fixture';

    // Boundary adapters — translate full CompiledXCatalog to inline structural inputs.
    // Matches CLAUDE.md rules-engine purity contract.
    const classInput = {
      classes: compiledClassCatalog.classes.map((c) => ({ id: c.id, skillPointsPerLevel: c.skillPointsPerLevel })),
    };
    const featInput = { classFeatLists: compiledFeatCatalog.classFeatLists };
    const raceInput = { races: compiledRaceCatalog.races.map((r) => ({ id: r.id })) };

    export function LevelEditorActionBar() {
      const progressionState = useLevelProgressionStore();
      const foundationState = useCharacterFoundationStore();
      const featState = useFeatStore();
      const skillState = useSkillStore();
      const setActiveLevel = useLevelProgressionStore((s) => s.setActiveLevel);
      const setExpandedLevel = useLevelProgressionStore((s) => s.setExpandedLevel);
      const setActiveLevelSubStep = useLevelProgressionStore((s) => s.setActiveLevelSubStep);

      const level = progressionState.activeLevel as ProgressionLevel;
      if (level === 16) return null;

      const completion = selectLevelCompletionState(
        progressionState,
        foundationState,
        featState,
        skillState,
        classInput,
        featInput,
        raceInput,
        level,
      );
      const { label, disabled } = computeAdvanceLabel(completion);

      const handleClick = () => {
        if (disabled) return;
        const next = (level + 1) as ProgressionLevel;
        // Atomic dispatch (mirrors Phase 12.3-02 LevelRail.onClick pattern)
        setActiveLevel(next);
        setExpandedLevel(next);
        setActiveLevelSubStep('class');
      };

      return (
        <footer className="level-editor__action-bar" data-testid="level-editor-action-bar">
          <NwnButton
            variant="primary"
            disabled={disabled}
            onClick={handleClick}
            data-testid={`advance-to-level-${level + 1}`}
          >
            {label}
          </NwnButton>
        </footer>
      );
    }
