**File to create:**
```typescript
// packages/data-extractor/src/lib/sentinel-regex.ts (NEW)
/**
 * BioWare + NWN EE sentinel conventions observed in 2DA LABEL column:
 *   - "DELETED" literal (Puerta snapshot: apps/planner/src/data/compiled-feats.ts sourceRow 385 + 487)
 *   - "DELETED_*" family (NWN Cut Feats wiki)
 *   - "PADDING" (CEP reserved-range convention)
 *   - "****" — cell-level only, already null-converted by two-da-parser.ts L131
 *   - "UNUSED" (SPEC R8 constraint)
 */
export const SENTINEL_REGEX =
  /^(\s*\*{3,}\s*(DELETED|UNUSED|PADDING)\s*\*{3,}\s*|\s*(DELETED|UNUSED|PADDING)(_.*)?\s*|\s*\*{3,}\s*)$/i;

export function isSentinelLabel(label: string | null | undefined): boolean {
  if (!label) return false;
  return SENTINEL_REGEX.test(label.trim());
}
```
