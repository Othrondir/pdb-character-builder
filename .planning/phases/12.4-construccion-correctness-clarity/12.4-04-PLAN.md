```typescript
// apps/planner/src/features/level-progression/selectors.ts (ADDITIONS)
export function isClaseLevelComplete(
  progressionState: LevelProgressionStoreState,
  level: ProgressionLevel,
): boolean {
  return (progressionState.levels.find((r) => r.level === level)?.classId ?? null) !== null;
}

export function isHabilidadesLevelComplete(
  progressionState: LevelProgressionStoreState,
  foundationState: CharacterFoundationStoreState,
  featState: FeatStoreState,
  skillState: SkillStoreState,
  classCatalog: ClassCatalogInput,
  featCatalog: FeatCatalogInput,
  raceCatalog: RaceCatalogInput,
  level: ProgressionLevel,
): boolean {
  // Build the BuildSnapshot shape per-level-budget expects, then return skillPoints.remaining === 0
  const budget = computePerLevelBudget(snapshot, level, classCatalog, featCatalog, raceCatalog);
  return budget.skillPoints.remaining === 0;
}

export function isDotesLevelComplete(/* same args */, level): boolean {
  const budget = computePerLevelBudget(snapshot, level, classCatalog, featCatalog, raceCatalog);
  return budget.featSlots.chosen === budget.featSlots.total && budget.featSlots.total > 0;
  // Note: total > 0 guards the edge case where a level has no class assigned yet
  // (budget.featSlots.total === 0 → don't say "complete", say "neutral").
}
```

    ```typescript
    import {
      computePerLevelBudget,
      type BuildSnapshot,
      type ClassCatalogInput,
      type FeatCatalogInput,
      type RaceCatalogInput,
    } from '@rules-engine/progression/per-level-budget';
    import type { CharacterFoundationStoreState } from '@planner/features/character-foundation/store';
    import type { FeatStoreState } from '@planner/features/feats/store';
    import type { SkillStoreState } from '@planner/features/skills/store';

    export function isClaseLevelComplete(
      progressionState: LevelProgressionStoreState,
      level: ProgressionLevel,
    ): boolean {
      const record = progressionState.levels.find((r) => r.level === level);
      return (record?.classId ?? null) !== null;
    }

    function toBuildSnapshot(
      progressionState: LevelProgressionStoreState,
      foundationState: CharacterFoundationStoreState,
      featState: FeatStoreState,
      skillState: SkillStoreState,
    ): BuildSnapshot {
      return {
        raceId: foundationState.raceId ?? null,
        classByLevel: Object.fromEntries(progressionState.levels.map((r) => [r.level, r.classId])),
        abilityScores: { int: foundationState.baseAttributes?.int ?? 10 },
        intAbilityIncreasesBeforeLevel: (lvl) =>
          progressionState.levels
            .filter((r) => r.level < lvl && r.abilityIncrease === 'int')
            .length,
        chosenFeatIdsAtLevel: (lvl) => featState.featsByLevel?.[lvl] ?? [],
        spentSkillPointsAtLevel: (lvl) =>
          Object.values(skillState.skillRanksByLevel?.[lvl] ?? {}).reduce((sum, rank) => sum + Number(rank), 0),
      };
    }

    export function isHabilidadesLevelComplete(
      progressionState: LevelProgressionStoreState,
      foundationState: CharacterFoundationStoreState,
      featState: FeatStoreState,
      skillState: SkillStoreState,
      classCatalog: ClassCatalogInput,
      featCatalog: FeatCatalogInput,
      raceCatalog: RaceCatalogInput,
      level: ProgressionLevel,
    ): boolean {
      const snapshot = toBuildSnapshot(progressionState, foundationState, featState, skillState);
      const budget = computePerLevelBudget(snapshot, level, classCatalog, featCatalog, raceCatalog);
      // If no class at this level, skillPoints.budget is 0 — NOT "complete", but "neutral".
      if (budget.skillPoints.budget === 0) return false;
      return budget.skillPoints.remaining === 0;
    }

    export function isDotesLevelComplete(
      progressionState: LevelProgressionStoreState,
      foundationState: CharacterFoundationStoreState,
      featState: FeatStoreState,
      skillState: SkillStoreState,
      classCatalog: ClassCatalogInput,
      featCatalog: FeatCatalogInput,
      raceCatalog: RaceCatalogInput,
      level: ProgressionLevel,
    ): boolean {
    ```
