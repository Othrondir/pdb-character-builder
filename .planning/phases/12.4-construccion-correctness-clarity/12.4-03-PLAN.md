```typescript
// packages/rules-engine/src/progression/per-level-budget.ts

/**
 * Minimal structural catalog inputs. Lists ONLY the fields this selector reads.
 * Caller-side adapters (apps/planner) translate full CompiledXCatalog → these
 * minimal inputs at the boundary. This keeps rules-engine framework-agnostic
 * (CLAUDE.md: "pure TypeScript domain model + validation engine, no React imports"
 * — and by extension no cross-package extractor imports either).
 */
export interface ClassCatalogInput {
  classes: ReadonlyArray<{
    id: string;
    skillPointsPerLevel: number;
  }>;
}

export interface FeatCatalogInput {
  // Pass-through shape required by determineFeatSlots(classFeatLists)
  classFeatLists: Readonly<Record<string, ReadonlyArray<{
    featId: string;
    grantAtLevel?: number;
    isBonusSlot?: boolean;
  }>>>;
}

export interface RaceCatalogInput {
  // Reserved: currently the selector hardcodes Humano canon (A1/A2).
  // Kept as a parameter slot so future race-assembler enrichment can populate
  // `bonusFeatAtFirstLevel` / `extraSkillPointsPerLevel` per-race without API change.
  races?: ReadonlyArray<{ id: string }>;
}

export interface PerLevelBudget {
  featSlots: {
    general: number;      // 0 or 1
    classBonus: number;   // 0 or 1
    raceBonus: number;    // 0 or 1 (Humano L1)
    total: number;        // sum of the three
    chosen: number;       // feats the user has selected at this level
    remaining: number;    // total - chosen
  };
  skillPoints: {
    budget: number;       // full budget including L1 ×4 and Humano +1/level
    spent: number;        // ranks allocated at this level (sum across skills)
    remaining: number;    // budget - spent
  };
}

export interface BuildSnapshot {
  raceId: string | null;
  classByLevel: Record<number, string | null>;
  abilityScores: { int: number };
  intAbilityIncreasesBeforeLevel: (level: number) => number;
  chosenFeatIdsAtLevel: (level: number) => string[];
  spentSkillPointsAtLevel: (level: number) => number;
}

export function computePerLevelBudget(
  build: BuildSnapshot,
  level: number,
  classCatalog: ClassCatalogInput,
  featCatalog: FeatCatalogInput,
  raceCatalog: RaceCatalogInput,
): PerLevelBudget;
```

```typescript
import type { ClassCatalogInput, FeatCatalogInput, RaceCatalogInput } from '@rules-engine/progression/per-level-budget';
import { compiledClassCatalog } from '@planner/data/compiled-classes';
import { compiledFeatCatalog } from '@planner/data/compiled-feats';
import { compiledRaceCatalog } from '@planner/data/compiled-races';

export function toClassCatalogInput(): ClassCatalogInput {
  return {
    classes: compiledClassCatalog.classes.map((c) => ({
      id: c.id,
      skillPointsPerLevel: c.skillPointsPerLevel,
    })),
  };
}
export function toFeatCatalogInput(): FeatCatalogInput {
  return { classFeatLists: compiledFeatCatalog.classFeatLists };
}
export function toRaceCatalogInput(): RaceCatalogInput {
  return { races: compiledRaceCatalog.races.map((r) => ({ id: r.id })) };
}
```

    ```typescript
    export function computePerLevelBudget(
      build: BuildSnapshot,
      level: number,
      classCatalog: ClassCatalogInput,
      featCatalog: FeatCatalogInput,
      _raceCatalog: RaceCatalogInput,
    ): PerLevelBudget {
      const classId = build.classByLevel[level] ?? null;

      if (classId === null) {
        return {
          featSlots: { general: 0, classBonus: 0, raceBonus: 0, total: 0, chosen: 0, remaining: 0 },
          skillPoints: { budget: 0, spent: 0, remaining: 0 },
        };
      }

      const classRow = classCatalog.classes.find((c) => c.id === classId) ?? null;
      const classLevelInClass = Object.entries(build.classByLevel)
        .filter(([lvl, cid]) => Number(lvl) <= level && cid === classId)
        .length;

      // `determineFeatSlots` signature accepts `classFeatLists` — FeatCatalogInput.classFeatLists
      // is structurally compatible with its existing `FeatCatalog['classFeatLists']` param.

      const general = slots.generalFeatSlot ? 1 : 0;
      const classBonus = slots.classBonusFeatSlot ? 1 : 0;
      const raceBonus = build.raceId === HUMAN_RACE_ID && level === 1 ? HUMAN_BONUS_FEAT_AT_L1 : 0;
      const total = general + classBonus + raceBonus;
      const chosen = build.chosenFeatIdsAtLevel(level).length;

      const intMod = Math.floor(
        (build.abilityScores.int + build.intAbilityIncreasesBeforeLevel(level) - 10) / 2,
      );
      const classSkillBase = classRow?.skillPointsPerLevel ?? 2;
      const base = Math.max(1, classSkillBase + intMod);
      const humanSkillBonus = build.raceId === HUMAN_RACE_ID ? HUMAN_SKILL_POINT_PER_LEVEL : 0;

      const spent = build.spentSkillPointsAtLevel(level);

      return {
        featSlots: { general, classBonus, raceBonus, total, chosen, remaining: total - chosen },
        skillPoints: { budget, spent, remaining: budget - spent },
      };
    }
    ```
