    ```tsx
    import { KeyboardEvent as ReactKeyboardEvent } from 'react';
    import { shellCopyEs } from '@planner/lib/copy/es';
    import type { FeatFamilyView } from './selectors';

    export interface FeatFamilyExpanderProps {
      family: FeatFamilyView;
      onSelectTarget: (featId: string) => void;
      onClose: () => void;
    }

    export function FeatFamilyExpander({ family, onSelectTarget, onClose }: FeatFamilyExpanderProps) {
      const handleKeyDown = (e: ReactKeyboardEvent<HTMLFieldSetElement>) => {
        if (e.key === 'Escape') {
          e.preventDefault();
          onClose();
        }
      };
      return (
        <fieldset
          className="feat-family-expander"
          onKeyDown={handleKeyDown}
          tabIndex={-1}
        >
          <legend className="feat-family-expander__legend">
            {shellCopyEs.feats.familyLegendPrefix} {family.paramLabel}
          </legend>
          <div className="feat-family-expander__options">
            {family.targets.map((target) => {
              const disabled = target.rowState !== 'selectable';
              return (
                <label key={target.featId} className="feat-family-expander__option">
                  <input
                    type="radio"
                    name={`family-${family.canonicalId}`}
                    disabled={disabled}
                    onChange={() => onSelectTarget(target.featId)}
                  />
                  <span>{target.label}</span>
                  {target.blockedReason?.pillLabel ? (
                    <span className="feat-picker__pill">{target.blockedReason.pillLabel}</span>
                  ) : null}
                </label>
              );
            })}
          </div>
        </fieldset>
      );
    }
    ```
