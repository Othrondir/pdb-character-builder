    Step 1 — Create `packages/data-extractor/src/lib/sentinel-regex.ts` with:

    ```typescript
    /**
     * BioWare + NWN EE sentinel conventions observed in 2DA LABEL column:
     *   - "DELETED" literal (Puerta snapshot: compiled-feats.ts sourceRow 385 + 487)
     *   - "DELETED_*" family (NWN Cut Feats wiki: https://nwn.wiki/spaces/NWN1/pages/38175424/Cut+Feats)
     *   - "PADDING" (CEP reserved-range convention)
     *   - "****" (3 or more stars — cell-level already null-converted by two-da-parser.ts L131,
     *     but LABEL-column occurrence must also be filtered)
     *   - "UNUSED" (SPEC R8 constraint)
     */
    export const SENTINEL_REGEX =
      /^(\s*\*{3,}\s*(DELETED|UNUSED|PADDING)\s*\*{3,}\s*|\s*(DELETED|UNUSED|PADDING)(_.*)?\s*|\s*\*{3,}\s*)$/i;

    export function isSentinelLabel(label: string | null | undefined): boolean {
      if (label === null || label === undefined) return false;
      const trimmed = label.trim();
      if (trimmed.length === 0) return false;
      return SENTINEL_REGEX.test(trimmed);
    }
    ```
