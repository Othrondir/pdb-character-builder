**Reference pattern — race fixture dedupe helper** (apps/planner/src/features/character-foundation/foundation-fixture.ts lines 115-126):
```typescript
function dedupeByCanonicalId<T extends { id: CanonicalId }>(entries: T[]): T[] {
  const seen = new Set<CanonicalId>();
  const unique: T[] = [];
  for (const entry of entries) {
    if (seen.has(entry.id)) {
      continue;
    }
    seen.add(entry.id);
    unique.push(entry);
  }
  return unique;
}
```
Mirror this exactly for classes, with one small addition: log dropped entries via `console.warn` alongside the sourceRow so the extractor backlog stays traceable (the race version is silent; the class version is not because the sourceRow index is useful diagnostic signal for the extractor fix).

     * class:harper-54). Tracked in 12.2-CONTEXT.md `<deferred>`.

    function dedupeCompiledClassesByCanonicalId(
      entries: readonly CompiledClass[],
    ): CompiledClass[] {
      const seen = new Set<string>();
      const unique: CompiledClass[] = [];
      for (const entry of entries) {

        }
        seen.add(entry.id);
        unique.push(entry);
      }
      return unique;
    }
