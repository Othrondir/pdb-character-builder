```typescript
import { create } from 'zustand';
import type { CanonicalId } from '@rules-engine/contracts/canonical-id';

import { compiledSpellCatalog } from './compiled-magic-catalog';
import { PROGRESSION_LEVELS, type ProgressionLevel } from '../level-progression/progression-fixture';

export interface SwapRecord {
  appliedAtLevel: ProgressionLevel;
  forgotten: CanonicalId;           // spell:id removed from known list
  learned: CanonicalId;             // spell:id added in its place
}

export interface MagicLevelRecord {
  level: ProgressionLevel;
  domains: CanonicalId[];                           // cleric L1 only; empty otherwise
  spellbookAdditions: Record<number, CanonicalId[]>;   // spellLevel 0..9 -> spellIds (wizard)
  knownSpells: Record<number, CanonicalId[]>;          // spellLevel 0..9 -> spellIds (sorcerer, bard)
  swapsApplied: SwapRecord[];                        // non-empty only on swap-cadence levels
}

export interface MagicStoreState {
  activeLevel: ProgressionLevel;
  datasetId: string;
  lastEditedLevel: ProgressionLevel | null;
  levels: MagicLevelRecord[];

  setActiveLevel: (level: ProgressionLevel) => void;
  setDomains: (level: ProgressionLevel, domains: CanonicalId[]) => void;
  addSpellbookEntry: (level: ProgressionLevel, spellLevel: number, spellId: CanonicalId) => void;
  removeSpellbookEntry: (level: ProgressionLevel, spellLevel: number, spellId: CanonicalId) => void;
  addKnownSpell: (level: ProgressionLevel, spellLevel: number, spellId: CanonicalId) => void;
  removeKnownSpell: (level: ProgressionLevel, spellLevel: number, spellId: CanonicalId) => void;
  applySwap: (level: ProgressionLevel, forgotten: CanonicalId, learned: CanonicalId) => void;
  resetLevel: (level: ProgressionLevel) => void;
  resetMagicSelections: () => void;
}

export function createEmptyMagicLevels(): MagicLevelRecord[] {
  return PROGRESSION_LEVELS.map((level) => ({
    level,
    domains: [],
    spellbookAdditions: {},
    knownSpells: {},
    swapsApplied: [],
  }));
}

export function createInitialMagicState() {
  return {
    activeLevel: 1 as ProgressionLevel,
    datasetId: compiledSpellCatalog.datasetId,
    lastEditedLevel: null as ProgressionLevel | null,
    levels: createEmptyMagicLevels(),
  };
}

function updateLevel(
  levels: MagicLevelRecord[],
  level: ProgressionLevel,
  mut: (rec: MagicLevelRecord) => MagicLevelRecord,
): MagicLevelRecord[] {
  return levels.map((r) => (r.level === level ? mut(r) : r));
}

export const useMagicStore = create<MagicStoreState>((set) => ({
  ...createInitialMagicState(),

  setActiveLevel: (activeLevel) => set({ activeLevel }),

  setDomains: (level, domains) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, (r) => ({ ...r, domains: [...domains] })),
    })),

  addSpellbookEntry: (level, spellLevel, spellId) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, (r) => {
        const existing = r.spellbookAdditions[spellLevel] ?? [];
        if (existing.includes(spellId)) return r;
        return { ...r, spellbookAdditions: { ...r.spellbookAdditions, [spellLevel]: [...existing, spellId] } };
      }),
    })),

  removeSpellbookEntry: (level, spellLevel, spellId) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, (r) => {
        const existing = r.spellbookAdditions[spellLevel] ?? [];
        const filtered = existing.filter((id) => id !== spellId);
        return { ...r, spellbookAdditions: { ...r.spellbookAdditions, [spellLevel]: filtered } };
      }),
    })),

  addKnownSpell: (level, spellLevel, spellId) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, (r) => {
        const existing = r.knownSpells[spellLevel] ?? [];
        if (existing.includes(spellId)) return r;
        return { ...r, knownSpells: { ...r.knownSpells, [spellLevel]: [...existing, spellId] } };
      }),
    })),

  removeKnownSpell: (level, spellLevel, spellId) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, (r) => {
        const existing = r.knownSpells[spellLevel] ?? [];
        const filtered = existing.filter((id) => id !== spellId);
        return { ...r, knownSpells: { ...r.knownSpells, [spellLevel]: filtered } };
      }),
    })),

  applySwap: (level, forgotten, learned) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, (r) => ({
        ...r,
        swapsApplied: [...r.swapsApplied, { appliedAtLevel: level, forgotten, learned }],
      })),
    })),

  resetLevel: (level) =>
    set((state) => ({
      lastEditedLevel: level,
      levels: updateLevel(state.levels, level, () => ({ level, domains: [], spellbookAdditions: {}, knownSpells: {}, swapsApplied: [] })),
    })),

  resetMagicSelections: () => set(createInitialMagicState()),
}));
```

**File 3: `tests/phase-07/magic-store.spec.ts`** — store unit tests:

```typescript
import { describe, expect, it, beforeEach } from 'vitest';
import { useMagicStore, createInitialMagicState, createEmptyMagicLevels } from '@planner/features/magic/store';
import { compiledSpellCatalog } from '@planner/data/compiled-spells';

describe('phase 07 magic store', () => {
  beforeEach(() => {
    useMagicStore.setState(createInitialMagicState());
  });

  it('createEmptyMagicLevels returns 16 records', () => {
    const levels = createEmptyMagicLevels();
    expect(levels).toHaveLength(16);
    expect(levels[0]).toMatchObject({ level: 1, domains: [], spellbookAdditions: {}, knownSpells: {}, swapsApplied: [] });
  });

  it('createInitialMagicState wires datasetId from compiled catalog', () => {
    const initial = createInitialMagicState();
    expect(initial.datasetId).toBe(compiledSpellCatalog.datasetId);
    expect(initial.activeLevel).toBe(1);
    expect(initial.lastEditedLevel).toBeNull();
  });

  it('setDomains updates only the target level', () => {
    useMagicStore.getState().setDomains(1, ['domain:air' as any, 'domain:fire' as any]);
    const state = useMagicStore.getState();
    expect(state.levels[0].domains).toEqual(['domain:air', 'domain:fire']);
    expect(state.levels[1].domains).toEqual([]);
    expect(state.lastEditedLevel).toBe(1);
  });

  it('addSpellbookEntry dedupes by spellId per spellLevel', () => {
    useMagicStore.getState().addSpellbookEntry(1, 1, 'spell:misil-magico' as any);
    useMagicStore.getState().addSpellbookEntry(1, 1, 'spell:misil-magico' as any);
    expect(useMagicStore.getState().levels[0].spellbookAdditions[1]).toEqual(['spell:misil-magico']);
  });

  it('removeSpellbookEntry removes the target spell', () => {
    useMagicStore.getState().addSpellbookEntry(1, 1, 'spell:a' as any);
    useMagicStore.getState().addSpellbookEntry(1, 1, 'spell:b' as any);
    useMagicStore.getState().removeSpellbookEntry(1, 1, 'spell:a' as any);
    expect(useMagicStore.getState().levels[0].spellbookAdditions[1]).toEqual(['spell:b']);
  });

  it('addKnownSpell + removeKnownSpell', () => {
    useMagicStore.getState().addKnownSpell(1, 0, 'spell:luz' as any);
    expect(useMagicStore.getState().levels[0].knownSpells[0]).toEqual(['spell:luz']);
    useMagicStore.getState().removeKnownSpell(1, 0, 'spell:luz' as any);
    expect(useMagicStore.getState().levels[0].knownSpells[0]).toEqual([]);
  });

  it('applySwap appends a SwapRecord at the target level', () => {
    useMagicStore.getState().applySwap(4, 'spell:a' as any, 'spell:b' as any);
    const rec = useMagicStore.getState().levels[3].swapsApplied;
    expect(rec).toHaveLength(1);
    expect(rec[0]).toMatchObject({ appliedAtLevel: 4, forgotten: 'spell:a', learned: 'spell:b' });
  });

  it('resetLevel clears all state for the target level', () => {
```

```typescript
/**
 * Phase 7 Pitfall 4 fixed compute order — this function is magic pass 1.
 * It computes a BuildStateAtLevel without touching feat selections; feat selections
 * are layered in via selectedFeatIds from the feat store, but feat prereq re-evaluation
 * uses the updated casterLevelByClass.
 */
function computeMagicBuildStateAtLevel(
  level: ProgressionLevel,
  foundationState: CharacterFoundationStoreState,
  progressionState: LevelProgressionStoreState,
  skillState: SkillStoreState,
  featState: FeatStoreState,
): BuildStateAtLevel {
  const abilityScores: Record<string, number> = { ...foundationState.baseAttributes };
  const classLevels: Record<string, number> = {};
  for (const rec of progressionState.levels) {
    if (rec.level <= level) {
      if (rec.abilityIncrease) abilityScores[rec.abilityIncrease] = (abilityScores[rec.abilityIncrease] ?? 0) + 1;
      if (rec.classId) classLevels[rec.classId] = (classLevels[rec.classId] ?? 0) + 1;
    }
  }
  const bab = computeTotalBab(classLevels, compiledClassCatalog);
  const fortitudeSave = computeFortSave(classLevels, compiledClassCatalog);
  const skillRanks: Record<string, number> = {};
  for (const skillLevel of skillState.levels) {
    if (skillLevel.level <= level) {
      for (const alloc of skillLevel.allocations) {
        skillRanks[alloc.skillId] = (skillRanks[alloc.skillId] ?? 0) + alloc.rank;
      }
    }
  }
  const selectedFeatIds = new Set<string>();
  for (const featLevel of featState.levels) {
    if (featLevel.level < level) {   // strict: current level's feats not yet visible
      if (featLevel.classFeatId) selectedFeatIds.add(featLevel.classFeatId);
      if (featLevel.generalFeatId) selectedFeatIds.add(featLevel.generalFeatId);
    }
  }
  const casterLevelByClass = computeCasterLevelByClass(classLevels, compiledClassCatalog);
  return { abilityScores, bab, characterLevel: level, classLevels, fortitudeSave, selectedFeatIds, skillRanks, casterLevelByClass };
}

function getActiveClassAtLevel(level: ProgressionLevel, progressionState: LevelProgressionStoreState): CanonicalId | null {
  const rec = progressionState.levels.find((r) => r.level === level);
  return rec?.classId ?? null;
}

function dispatchParadigm(classId: CanonicalId | null, characterLevel: ProgressionLevel): MagicParadigm {
  if (!classId) return 'empty';
  switch (classId) {
    case 'class:cleric': return characterLevel === 1 ? 'domains' : 'prepared-summary';
    case 'class:wizard': return 'spellbook';
    case 'class:sorcerer':
    case 'class:bard': return 'known';
    case 'class:druid':
    case 'class:paladin':
    case 'class:ranger': return 'prepared-summary';
    default: return 'empty';
  }
}

function isSwapLevel(classId: CanonicalId | null, level: ProgressionLevel): boolean {
  if (classId === 'class:sorcerer') return SORCERER_SWAP_LEVELS.has(level);
  if (classId === 'class:bard') return BARD_SWAP_LEVELS.has(level);
  return false;
}

function buildSlotsByLevel(
  classId: CanonicalId | null,
  casterLevelByClass: Record<string, number>,
  selectedCounts: Record<number, number>,
): Record<number, SlotStatus> {
  const out: Record<number, SlotStatus> = {};
  if (!classId) return out;
  const casterLevel = casterLevelByClass[classId] ?? 0;
  if (casterLevel < 1) return out;
  for (let spellLevel = 0; spellLevel <= 9; spellLevel++) {
    const max = computeSpellSlots(classId, casterLevel, spellLevel, compiledSpellCatalog);
    const current = selectedCounts[spellLevel] ?? 0;
    const status: SlotStatus['status'] = current > max ? 'repair_needed' : 'legal';
    out[spellLevel] = { current, max, status };
  }
  return out;
}
```

```typescript
export function selectMagicBoardView(
  magicState: MagicStoreState,
  featState: FeatStoreState,
  skillState: SkillStoreState,
  progressionState: LevelProgressionStoreState,
  foundationState: CharacterFoundationStoreState,
): MagicBoardView {
  const activeLevel = magicState.activeLevel;

  // Empty-progression guard
  const progressionHasClass = progressionState.levels.some((lvl) => lvl.classId !== null);
  if (!progressionHasClass) {
    return {
      activeSheet: {
        paradigm: 'empty', level: activeLevel, activeSpellLevel: 0,
        classId: null, classLabel: null, slotsByLevel: {}, eligibleSpells: [], selectedSpells: [],
        eligibleDomains: [], selectedDomains: [],
        emptyMessage: '', status: 'pending', swapAvailable: false,
        title: '',
      },
      emptyStateBody: 'La magia sigue bloqueada',   // replaced with shellCopyEs.magic.emptyStateBody in Plan 07-03 consumer
    };
  }

  const classId = getActiveClassAtLevel(activeLevel, progressionState);
  const paradigm = dispatchParadigm(classId, activeLevel);
  const buildState = computeMagicBuildStateAtLevel(activeLevel, foundationState, progressionState, skillState, featState);

  // Run revalidation for the active level
  const revalidationInput: MagicLevelInput[] = magicState.levels.map((lvl) => ({
    buildState: computeMagicBuildStateAtLevel(lvl.level as ProgressionLevel, foundationState, progressionState, skillState, featState),
    level: lvl.level,
    domainsSelected: lvl.domains,
    spellbookAdditions: lvl.spellbookAdditions,
    knownSpells: lvl.knownSpells,
    swapsApplied: lvl.swapsApplied,
  }));
  const revalidated = revalidateMagicSnapshotAfterChange({
    levels: revalidationInput, spellCatalog: compiledSpellCatalog, domainCatalog: compiledDomainCatalog,
  });
  const activeRev = revalidated.find((r) => r.level === activeLevel) ?? { level: activeLevel, status: 'pending' as MagicEvaluationStatus, issues: [], inheritedFromLevel: null };

  // Active record
  const activeRecord = magicState.levels.find((r) => r.level === activeLevel)!;

  // Class label
  const classDef = classId ? compiledClassCatalog.classes.find((c) => c.id === classId) : null;
  const classLabel = classDef?.label ?? null;

  // Choose activeSpellLevel: find the lowest spellLevel with max > 0, or 0
  let activeSpellLevel = 0;
  if (paradigm === 'spellbook' || paradigm === 'known') {
    const casterLevel = buildState.casterLevelByClass[classId!] ?? 0;
    for (let sl = 0; sl <= 9; sl++) {
      if (computeSpellSlots(classId!, casterLevel, sl, compiledSpellCatalog) > 0) { activeSpellLevel = sl; break; }
    }
  }

  // slotsByLevel with current = number of selections at that level
  const selectedCounts: Record<number, number> = {};
  if (paradigm === 'spellbook') {
    for (const [sl, list] of Object.entries(activeRecord.spellbookAdditions)) selectedCounts[Number(sl)] = list.length;
  } else if (paradigm === 'known') {
    for (const [sl, list] of Object.entries(activeRecord.knownSpells)) selectedCounts[Number(sl)] = list.length;
  }
  const slotsByLevel = buildSlotsByLevel(classId, buildState.casterLevelByClass, selectedCounts);

  // Eligible spells for current paradigm + active spell level
  let eligibleSpells: SpellOptionView[] = [];
  let selectedSpells: SpellOptionView[] = [];
  if ((paradigm === 'spellbook' || paradigm === 'known') && classId) {
    const casterLevel = buildState.casterLevelByClass[classId] ?? 0;
    const alreadyKnownSet = new Set<string>();
    if (paradigm === 'spellbook') {
      for (const list of Object.values(activeRecord.spellbookAdditions)) for (const id of list) alreadyKnownSet.add(id);
    } else {
      for (const list of Object.values(activeRecord.knownSpells)) for (const id of list) alreadyKnownSet.add(id);
    }
    const elig = getEligibleSpellsAtLevel({
      classId, casterLevel, spellLevel: activeSpellLevel, catalog: compiledSpellCatalog, alreadyKnown: alreadyKnownSet,
    });
    const selectedIdsAtLevel = new Set<string>(
      paradigm === 'spellbook'
        ? (activeRecord.spellbookAdditions[activeSpellLevel] ?? [])
        : (activeRecord.knownSpells[activeSpellLevel] ?? [])
    );
    const slotStatus = slotsByLevel[activeSpellLevel]?.status ?? 'legal';
    const slotOverflow = slotStatus === 'repair_needed';
    eligibleSpells = elig.eligible.map((sp) => {
      const missing = detectMissingSpellData(sp.id, compiledSpellCatalog) !== null;
      const prereq = evaluateSpellPrerequisites(sp, buildState, compiledSpellCatalog);
      const isSelected = selectedIdsAtLevel.has(sp.id);
      return {
        spellId: sp.id as CanonicalId,
        label: sp.label,
        schoolLabel: SCHOOL_LABELS_ES[sp.school] ?? SCHOOL_LABELS_ES.unknown,
        description: sp.description,
        selected: isSelected,
        blocked: !prereq.met || (!isSelected && slotOverflow),
        blockReason: !prereq.met ? prereq.checks.filter((c) => !c.met).map((c) => `${c.label} ${c.required}`).join(', ') : (slotOverflow && !isSelected ? 'Ranura agotada' : null),
        missingData: missing,
      };
    });
    // Selected spells view
    const selectedSpellIds = paradigm === 'spellbook'
      ? Object.values(activeRecord.spellbookAdditions).flat()
      : Object.values(activeRecord.knownSpells).flat();
    selectedSpells = selectedSpellIds.map((id) => {
      const sp = compiledSpellCatalog.spells.find((s) => s.id === id);
      if (!sp) return null;
      const missing = detectMissingSpellData(sp.id, compiledSpellCatalog) !== null;
      return {
        spellId: sp.id as CanonicalId,
        label: sp.label,
        schoolLabel: SCHOOL_LABELS_ES[sp.school] ?? SCHOOL_LABELS_ES.unknown,
        description: sp.description,
        selected: true,
        blocked: false,
        blockReason: null,
        missingData: missing,
      };
    }).filter((x): x is SpellOptionView => x !== null);
  }

  // Domains (cleric L1 only)
  let eligibleDomains: DomainOptionView[] = [];
  let selectedDomains: DomainOptionView[] = [];
  if (paradigm === 'domains') {
    const selectedDomainIds = new Set<string>(activeRecord.domains);
    const capReached = activeRecord.domains.length >= MAX_DOMAINS_PER_CLERIC;
    eligibleDomains = compiledDomainCatalog.domains.map((d) => {
      const missing = detectMissingDomainData(d.id, compiledDomainCatalog) !== null;
      const result = evaluateDomainSelection(d.id, activeRecord.domains, buildState, compiledDomainCatalog);
      const isSelected = selectedDomainIds.has(d.id);
      const grantedFeatLabels = d.grantedFeatIds.map((fid) => {
        const feat = compiledFeatCatalog.feats.find((f) => f.id === fid);
        return feat?.label ?? fid;
      });
      const bonusSpellLabels: Record<number, string[]> = {};
      for (const [sl, spellIds] of Object.entries(d.spellIds)) {
        bonusSpellLabels[Number(sl)] = (spellIds ?? []).map((sid) => {
          const sp = compiledSpellCatalog.spells.find((s) => s.id === sid);
          return sp?.label ?? sid;
        });
      }
      return {
        domainId: d.id as CanonicalId,
        label: d.label,
        description: d.description,
        grantedFeatLabels, bonusSpellLabels,
        selected: isSelected,
        blocked: !result.met && !isSelected,
        blockReason: !result.met && !isSelected ? (capReached ? `Máximo ${MAX_DOMAINS_PER_CLERIC} dominios` : result.checks.filter((c) => !c.met).map((c) => c.label).join(', ')) : null,
        missingData: missing,
      };
    });
    selectedDomains = eligibleDomains.filter((d) => d.selected);
  }

  return {
    activeSheet: {
      paradigm, level: activeLevel, activeSpellLevel,
      classId, classLabel, slotsByLevel, eligibleSpells, selectedSpells,
      eligibleDomains, selectedDomains,
      emptyMessage: '',
      status: activeRev.status,
      swapAvailable: isSwapLevel(classId, activeLevel),
      title: '',   // filled by consumer via shellCopyEs.magic.*StepTitle (Plan 07-03)
    },
    emptyStateBody: null,
  };
}

export function selectMagicSheetTabView(
  magicState: MagicStoreState,
  featState: FeatStoreState,
  skillState: SkillStoreState,
  progressionState: LevelProgressionStoreState,
  foundationState: CharacterFoundationStoreState,
) {
  // Groups by class+spellLevel, shows selected spells + domain bonus spells
  // Full implementation mirrors selectFeatSheetTabView shape.

  let totalCount = 0;
  let invalidCount = 0;
  for (const lvl of magicState.levels) {
    const classId = getActiveClassAtLevel(lvl.level as ProgressionLevel, progressionState);
    if (!classId) continue;
    const classDef = compiledClassCatalog.classes.find((c) => c.id === classId);
    if (!classDef?.spellCaster) continue;
    const casterLevelByClass = computeCasterLevelByClass(
      Object.fromEntries(progressionState.levels.filter((r) => r.level <= lvl.level && r.classId).map((r) => [r.classId!, progressionState.levels.filter((x) => x.level <= lvl.level && x.classId === r.classId).length])),
      compiledClassCatalog,
    );
    const casterLevel = casterLevelByClass[classId] ?? 0;
    if (casterLevel < 1) continue;
    const spells: typeof groups[number]['spells'] = [];
    // Spellbook additions
    for (const [slKey, list] of Object.entries(lvl.spellbookAdditions)) {
      for (const spellId of list) {
        const sp = compiledSpellCatalog.spells.find((s) => s.id === spellId);
        if (sp) { spells.push({ spellId, label: sp.label, slot: 'spellbook', status: 'legal', statusReason: null }); totalCount++; }
      }
    }
    for (const [slKey, list] of Object.entries(lvl.knownSpells)) {
      for (const spellId of list) {
        const sp = compiledSpellCatalog.spells.find((s) => s.id === spellId);
        if (sp) { spells.push({ spellId, label: sp.label, slot: 'known', status: 'legal', statusReason: null }); totalCount++; }
      }
    }
    if (spells.length > 0) {
      groups.push({ heading: `${classDef.label} - Nivel de lanzador ${casterLevel}`, level: lvl.level, spells });
    }
  }
  return { groups, invalidCount, totalCount };
}

export function selectMagicSummary(
  magicState: MagicStoreState,
  featState: FeatStoreState,
  skillState: SkillStoreState,
  progressionState: LevelProgressionStoreState,
  foundationState: CharacterFoundationStoreState,
): MagicSummaryView {
  const revalInput: MagicLevelInput[] = magicState.levels.map((lvl) => ({
    buildState: computeMagicBuildStateAtLevel(lvl.level as ProgressionLevel, foundationState, progressionState, skillState, featState),
    level: lvl.level, domainsSelected: lvl.domains,
    spellbookAdditions: lvl.spellbookAdditions, knownSpells: lvl.knownSpells, swapsApplied: lvl.swapsApplied,
  }));
  const rev = revalidateMagicSnapshotAfterChange({ levels: revalInput, spellCatalog: compiledSpellCatalog, domainCatalog: compiledDomainCatalog });
  const blockedLevels = rev.filter((r) => r.status === 'blocked' || r.status === 'illegal').map((r) => r.level);
  const highestConfiguredLevel = magicState.levels
    .filter((lvl) => lvl.domains.length > 0 || Object.values(lvl.spellbookAdditions).some((l) => l.length > 0) || Object.values(lvl.knownSpells).some((l) => l.length > 0))
    .reduce((max, lvl) => Math.max(max, lvl.level), 0);
  // summaryStatus = worst status across levels
  const summaryStatus: MagicEvaluationStatus = rev.reduce<MagicEvaluationStatus>(
    (worst, r) => (STATUS_ORDER[r.status] < STATUS_ORDER[worst] ? r.status : worst),
    'pending',
  );
  // planState string key (Plan 07-03 maps to shellCopyEs.magic.planStates):
  const planState = highestConfiguredLevel === 0 ? 'empty'
    : summaryStatus === 'illegal' || summaryStatus === 'blocked' ? 'repair'
    : summaryStatus === 'pending' ? 'inProgress' : 'ready';
  return { blockedLevels, highestConfiguredLevel, planState, summaryStatus };
}
```
