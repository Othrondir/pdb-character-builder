    ```tsx
    // Phase 12.8-03 (D-04, UAT-2026-04-23 F3) — auto-scroll to the general
    // section when the class-bonus slot fills. Without this the viewport
    // stays on the class section and the user's second click lands on
    // class rows again (overwrites the first pick — UAT evidence counter
    // stuck at 1/2). Uses `block: 'nearest'` so wider viewports where the
    // general section is already visible do not get a gratuitous scroll.
    //
    // Guard: only trigger on an actual null → non-null transition. A dep
    // array of just [currentRecord?.classFeatId] would re-fire if the user
    // picks a DIFFERENT class feat (still non-null), which is correct but
    // we also want to avoid firing on initial mount when classFeatId is
    // already set (e.g., user navigated back to a completed level).
    const prevClassFeatIdRef = useRef<string | null | undefined>(undefined);
    useEffect(() => {
      const prev = prevClassFeatIdRef.current;
      const next = currentRecord?.classFeatId ?? null;
      // Fire only when class slot transitions from empty (null) to non-null.
      // Skip the initial-mount case where prev === undefined (we have not
      // yet observed a change).
      if (prev === null && next !== null) {
        // Defer one frame so the general section has time to re-render
        // with the updated "Ahora" pill + feat-sheet__group--current class.
        requestAnimationFrame(() => {
          const generalSection = document.querySelector<HTMLElement>(
            '[data-slot-section="general"]',
          );
          if (generalSection === null) {
            return;
          }
          const firstRow =
            generalSection.querySelector<HTMLElement>('button.feat-picker__row')
            ?? generalSection;
          firstRow.scrollIntoView({ block: 'nearest', behavior: 'smooth' });
        });
      }
      prevClassFeatIdRef.current = next;
    }, [currentRecord?.classFeatId]);
    ```
